package uk.co.ziazoo.snappyviewer.view
{
    import mx.collections.ArrayCollection;
    import mx.collections.IViewCursor;
    
    import uk.co.ziazoo.snappyviewer.data.IItem;

    public class AbstractItemView implements IItemView
    {
        protected var _items : ArrayCollection;
        protected var _watchCollection : Boolean;
        protected var _displayer : IItemDisplayer;
        
        public function AbstractItemView ()
        {
            _items = new ArrayCollection();
            _watchCollection = true;
        }
        
        public function set displayer( displayer : IItemDisplayer ) : void
        {
            _displayer = displayer;
        }
        
        [Bindable]
        public function set items(value:ArrayCollection) : void
        {
            _items = value;
        }
        
        public function get items () : ArrayCollection
        {
            return _items;
        }
        
        public function get cursor() : IViewCursor
        {
            return _items.createCursor();
        }
        
        public function getItemAt( index : int ) : IItem
        {
            return _items.getItemAt( index ) as IItem;
        }
        
        public function display () : void
        {
            
        }
        
        public function set watchCollection ( value : Boolean ) : void
        {
            _watchCollection = value;
        }
        
        public function addChangeListeners () : void
        {
            
        }
    }
}
