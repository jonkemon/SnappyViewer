package uk.co.ziazoo.snappyviewer.view.imageViewer
{
    import flash.display.DisplayObjectContainer;
    
    import mx.events.CollectionEvent;
    import mx.events.CollectionEventKind;
    
    import uk.co.ziazoo.snappyviewer.model.SnappyModelLocator;
    import uk.co.ziazoo.snappyviewer.view.AbstractItemView;
    import uk.co.ziazoo.snappyviewer.view.IItemDisplayer;
    import uk.co.ziazoo.snappyviewer.view.IItemView;

    public class ItemView extends AbstractItemView implements IItemView
    {
        private var _container : DisplayObjectContainer;
        
        public function ItemView( container : DisplayObjectContainer )
        {
            super();
            _container = container;
            addChangeListeners();
        }
        
        override public function display() : void 
        {
            super.display();
            _displayer.display( items );
        }
        
        private function addItemstoDisplay ( items : Array ) : void
        {            
            for ( var i : int = 0 ; i < _items.length ; i++ )
            {
                var item : Item = _items.getItemAt( i ) as Item;
                
                item.alpha = 0;
                
                // display the item
                _container.addChild( item );
            }
        }
            
        override public function addChangeListeners () : void
        {
            SnappyModelLocator.getInstance().iItemCollection.addEventListener(CollectionEvent.COLLECTION_CHANGE    , onDataChange );        
        }
        
        private function onDataChange ( event : CollectionEvent ) : void
        {
            // if items have been added to the data, add GridItems to represent them
            if ( event.kind == CollectionEventKind.ADD )
            {
                addItemstoDisplay( event.items );
                items = SnappyModelLocator.getInstance().iItemCollection;
            }
            
            // update the display to reflect the changes to the underlying data
            display();
        }
        
    }
}
