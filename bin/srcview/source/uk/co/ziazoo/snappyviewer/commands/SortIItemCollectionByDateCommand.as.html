package uk.co.ziazoo.snappyviewer.commands
{
    import com.adobe.cairngorm.commands.ICommand;
    import com.adobe.cairngorm.control.CairngormEvent;
    
    import mx.collections.ArrayCollection;
    import mx.collections.Sort;
    import mx.collections.SortField;
    
    import uk.co.ziazoo.snappyviewer.model.SnappyModelLocator;

    public class SortIItemCollectionByDateCommand implements ICommand
    {
        public function execute( event : CairngormEvent ):void
        {            
            var field : SortField = new SortField( "date" , false , false , true );
            
            var sort : Sort = new Sort();
            sort.fields = [ field ];
            
            var data : ArrayCollection = SnappyModelLocator.getInstance().iItemCollection; 
            
            data.sort = sort;
            data.refresh();
        }        
    }
}
